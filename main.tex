\section*{Problem 12.4.1}

Show that, for $|s\rangle = \cos(\theta / 2) | \uparrow \rangle +
\exp(i\phi)\sin(\theta / 2) | \downarrow \rangle$,

$
\langle s |\hat{\boldsymbol{\sigma}} | s \rangle =
\bm{i}\sin\theta\cos\phi +
\bm{j}\sin\theta\sin\phi +
\bm{k}\cos\theta
$.

\boxedanswer{
  \begin{align*}
    |s\rangle &=
    \begin{pmatrix}
      a \\ b \\
    \end{pmatrix} & a &= \cos\frac{\theta}{2} & b &=
    e^{i\phi}\sin\frac{\theta}{2} \\
    \sigma_x &=
    \begin{pmatrix}
      0 & 1 \\ 1 & 0 \\
    \end{pmatrix} &  \sigma_y &=
    \begin{pmatrix}
      0 & -i \\ i & 0 \\
    \end{pmatrix} & \sigma_z &=
    \begin{pmatrix}
      1 & 0 \\ 0 & -1 \\
    \end{pmatrix} \\
    \hat{\boldsymbol{\sigma}}_x |s\rangle &=
    \begin{pmatrix}
      b \\ a \\
    \end{pmatrix} &
    \langle s|\hat{\boldsymbol{\sigma}}_x|s\rangle &=
    a^*b + b^*a = 2\Re(a^*b) \\
    2\Re(a^*b) &= 2\cos\frac{\theta}{2}\sin\frac{\theta}{2}\cos\phi = \sin\theta\cos\phi \\
    \sigma_y|s\rangle &= \begin{pmatrix} -ib \\ ia \end{pmatrix} \\
    \langle s|\sigma_ys\rangle&=-ia^*b+ib*a = 2\Im(b^*a) \\
    2\Im(b^*a) &= 2\cos\frac\theta2\sin\theta2\sin\phi = \sin\theta\sin\phi \\
    \sigma_z|s\rangle&=\begin{pmatrix}a\\-b\end{pmatrix} & \langle s|\sigma_z|s\rangle&=|a|^2-|b|^2=\cos^2\frac\theta2-\sin^2\frac\theta2=\cos\theta \\
    \Aboxed{\langle s|\hat{\boldsymbol{\sigma}}|s\rangle&= \bm{i}\sin\theta\cos\phi+\bm{j}\sin\theta\sin\phi+\bm{k}\cos\theta}
  \end{align*}
}

\section*{Problem 12.4.2}

Suppose an electron is in the spin state
$|s\rangle = (1/\sqrt2)
[|\uparrow\rangle + i| \uparrow \rangle]
$, and that the electron spin magnetic moment operator can be written as
$\hat{\boldsymbol{\mu}}_e = g \mu_B\hat{\boldsymbol{\sigma}}$.
What is the expectation value of the electron spin magnetic moment?
[Note: the result is a vector].

\boxedanswer{
  \begin{align*}
      |s\rangle &= \frac{1}{\sqrt2}\begin{pmatrix}1\\i\end{pmatrix} \\
      \langle\sigma_i\rangle&=\langle s|\sigma_i|s\rangle \\
      \langle\sigma_x\rangle&=\psi^\dagger\sigma_x\psi=\frac{1}{2}\begin{pmatrix}1&-i\end{pmatrix}\begin{pmatrix}0&1\\1&0\end{pmatrix}\begin{pmatrix}1\\i\end{pmatrix}=0\\
      \langle\sigma_y\rangle&=\psi^\dagger\sigma_x\psi=\frac{1}{2}\begin{pmatrix}1&-i\end{pmatrix}\begin{pmatrix}0&-i\\i&0\end{pmatrix}\begin{pmatrix}1\\i\end{pmatrix}=1\\
      \langle\sigma_z\rangle&=\psi^\dagger\sigma_x\psi=\frac{1}{2}\begin{pmatrix}1&-i\end{pmatrix}\begin{pmatrix}1&0\\0&-1\end{pmatrix}\begin{pmatrix}1\\i\end{pmatrix}=0\\
      \langle\hat{\boldsymbol{\sigma}}_z\rangle&=\hat y \\
      \Aboxed{\langle\hat{\boldsymbol{\mu}}_e\rangle&=g\mu_B\hat y}
  \end{align*}
}

\section*{Problem 12.4.3}

Consider an arbitrary spin state
$|s\rangle = \cos(\theta/2)|\uparrow\rangle +
\exp(i\phi)\sin (\theta/2)|\downarrow\rangle$,
and operate on it with
the $\hat\sigma_z$ operator. Describe the result of this operation in terms of
a rotation of the spin polarization vector on the Bloch sphere.

\boxedanswer{
    \begin{align*}
        |s\rangle&=\cos\frac{\theta}{2}|\uparrow\rangle+e^{i\theta}\sin\frac\theta2|\downarrow\rangle\\
        \hat\sigma_z&=\begin{pmatrix}1&0\\0&-1\end{pmatrix}\\
        \hat\sigma_z|s\rangle&=\cos\frac\theta2|\uparrow-e^{i\theta}\sin\frac\theta2|\downarrow\rangle=\cos\frac\theta2|\uparrow\rangle+e^{i(\phi+\pi)}\sin\frac\theta2|\downarrow\rangle\\
        \bm{n}&=(\sin\theta\cos\phi,\sin\theta\sin\phi,\cos\theta)\\
        \Aboxed{\bm{n'}&=(\sin\theta\cos(\phi+\phi),\sin\theta\sin(\phi+\pi),\cos\theta) = 
        (-\sin\theta\cos\phi,-\sin\theta\sin\phi, \cos\theta)} \\
        (x,y,z) &\rightarrow(-x,-y,z)
    \end{align*}

    $\hat\sigma_z$ rotates by $\pi$ on the z-axis of the Bloch sphere, flipping the x and y components,
    but not changing the z axis.

}
