\section*{Problem 12.4.1}

Show that, for $|s\rangle = \cos(\theta / 2) | \uparrow \rangle +
\exp(i\phi)\sin(\theta / 2) | \downarrow \rangle$,

$
\langle s |\hat{\boldsymbol{\sigma}} | s \rangle =
\bm{i}\sin\theta\cos\phi +
\bm{j}\sin\theta\sin\phi +
\bm{k}\cos\theta
$.

\boxedanswer{
  \begin{align*}
    |s\rangle &=
    \begin{pmatrix}
      a \\ b \\
    \end{pmatrix} & a &= \cos\frac{\theta}{2} & b &=
    e^{i\phi}\sin\frac{\theta}{2} \\
    \sigma_x &=
    \begin{pmatrix}
      0 & 1 \\ 1 & 0 \\
    \end{pmatrix} &  \sigma_y &=
    \begin{pmatrix}
      0 & -i \\ i & 0 \\
    \end{pmatrix} & \sigma_z &=
    \begin{pmatrix}
      1 & 0 \\ 0 & -1 \\
    \end{pmatrix} \\
    \hat{\boldsymbol{\sigma}}_x |s\rangle &=
    \begin{pmatrix}
      b \\ a \\
    \end{pmatrix} &
    \langle s|\hat{\boldsymbol{\sigma}}_x|s\rangle &=
    a^*b + b^*a = 2\mathbb{R}(a^*b) \\
  \end{align*}
}

\section*{Problem 12.4.2}

Suppose an electron is in the spin state
$|s\rangle = (1/\sqrt2)
[|\uparrow\rangle + i| \uparrow \rangle]
$, and that the electron spin magnetic moment operator can be written as
$\hat{\boldsymbol{\mu}}_e = g \mu_B\hat{\boldsymbol{\sigma}}$.
What is the expectation value of the electron spin magnetic moment?
[Note: the result is a vector].

\boxedanswer{
  \begin{align*}
    1
  \end{align*}
}
